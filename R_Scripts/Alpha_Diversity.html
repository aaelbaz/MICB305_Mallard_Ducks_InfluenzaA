# Setup ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
# taxa must be columns instead of rows, this will convert it if needed
otu = as(otu_table(psrare), "matrix")
if(taxa_are_rows(psrare)==TRUE) otu <- t(otu)

# Extract your phyloseq object's tree
tree_faith = phy_tree(psrare)

# check that all the ASV names match between OTU table and tree
stopifnot(all(colnames(otu) %in% tree_faith$tip.label))
# reorder tree tips to match ASV order, if needed:
# Won't hurt to run this on your data if it's unnecessary.
#    remove any unmatched tips
tree_faith = drop.tip(tree_faith, setdiff(tree_faith$tip.label, colnames(otu)))

faith = pd(otu_pa, tree_faith, include.root = TRUE) %>% select(-SR)
