Introduction

Faith’s Diversity

Calculating the Faith’s PD values for each sample.

#Adapted from MICB 305
#Taxa converted from columns to rows
otu = as(otu_table(psrare), "matrix")
if(taxa_are_rows(psrare)==TRUE) otu <- t(otu)

# Extract your phyloseq object's tree
tree_faith = phy_tree(psrare)

#Check that all the ASV names match between OTU table and tree
stopifnot(all(colnames(otu) %in% tree_faith$tip.label))

#Remove any unmatched tips
tree_faith = drop.tip(tree_faith, setdiff(tree_faith$tip.label, colnames(otu)))
#Reorder to match
tree_faith = keep.tip(tree_faith, colnames(otu))

#Rooting the tree
if(!is.rooted(tree_faith)) tree_faith <- root(tree_faith, outgroup=tree_faith$tip.label[1], resolve.root=TRUE)

# Calculate Faith's PD for each sample 
#    1. Convert reads to presence/absence and pull into a matrix
otu_pa = otu # Make a copy of your data, then...
otu_pa[otu_pa>0] = 1 #...convert to presence/absence (1 for present, 0 for absent)
#    2. Calculate metric
faith = pd(otu_pa, tree_faith, include.root = TRUE) %>% select(-SR)

#Merging phyloseq data with the Faith's PD data. 

metadata_rarefied = data.frame(sample_data(psrare))

faith_metadata = merge(metadata_rarefied, faith, by= "row.names")
